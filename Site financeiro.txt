function cliente(){
    var moneyhora = 0.67; // é o valor do consumo em reais por kw/h em SP
    var solar = 1.10; // O indice de kw/h por luminosidade em SP é de 5,6; entretanto fizemos uma redução devido a variação climática e tambem as varias fases durante o dia
    var cash = solar * ace.value;
    var tte = (((wat.value / 1000) * qtd.value) * ace.value);
    var eco = ((tte - cash) * moneyhora) * mes.value;
    var tt = (tte * moneyhora) * mes.value;
    var percentual = 100 - ((eco * 100) / tt);
    
    calculo.innerHTML = `Com a FireFly Solutions, você gastaria somente R$${(eco).toFixed(2)}, economizando ${(percentual).toFixed(2)}%, Uau!`
}
